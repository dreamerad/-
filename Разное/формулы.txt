// Строковые функции
1. strpos($string, $str, 0) - Выводит позицию подстроки в строке по символьно
$string - строка, $str - подстрока, 0 - с какого символо начинается поиск
2. strcmp($str1, $str2) - Бинарно - безопасное сравнение строка(Учитывается регистр),
возвращает -1 если $str1 < $str2, 1 если $str1 > $str2, 0 если строки равны.
$str1 - 1ая строка, $str2 - 2ая строка
3. strcasecmp($str1, $str2) - Бинарно - безопасное сравнение строка(Не учитывается регистр),
возвращает -1 если $str1 < $str2, 1 если $str1 > $str2, 0 если строки равны.
$str1 - 1ая строка, $str2 - 2ая строка
4. substr($min, $max) - Обрезает строку с левой($min) и правой($max) стороны
$rest = substr("abcdef", 3);    // возвращает "def"
5. str_replace($var1, $var2, $var3) - Меняет $var2 на $var1 в $var3
6. str_iraplace($var1, $var2, $var3) - Меняет $var2 на $var1 в $var3(Без учёта регистра)
7. strtolower($str) - Перевод строки в нижкий регистр
8. strtoupper($str) - Перевод строки в верхний регистр
9. strip_tags($str, $st) - Удаляет из строки все теги (Защита от XSS и SQL Inj)
$str - строка, $st - Теги которые удалять не нужно(Необязательный)
// Массивы
1. asort($array) - Сортировка значений массива по алфавиту(для строки) 
и по возрастанию(для чисел)
2. arsort($array) - Сортировка значений массива по убыванию по алфавиту(для строки) 
и по убыванию(для чисел)
3. ksort($array) - Сортировка массива по ключу
4. krsort($array) - Сортировка массива по ключу в обратном порядке
5. array_reverse($array, true) - Возвращает массив в обратном порядке, при true 
переворачиваться будут только значения
6. natsort($array) - натуральная сортировка
7. natcasesort($array) - натуральная сортировка, без учета регистра
8. sort($array) - сортировка списка в порядке возрастания
8. rsort($array) - сортировка списка в порядке убывания
9. array_miltisort($array,const) - сортировка многомерного массива
const - Константа задающая порядок сортировки(НЕобязательно)
10. shuffle($array) - перемешивает списки
11. array_flip($array) - меняет местами ключи и значения
12. in_array($val, $array) - возвращает true если в массиве есть $val
13. array_count_values($array) - возвращает сколько раз каждое значение встречается в массиве 
возвращает ассоциативный массив в котором, ключ - элемент масива, а значение - кол-во повторений
14. array_merge($array1, $array2) - Склеивает два массива в один
15. array_slice($array, n) - Возвращает часть массива с n-ого элемента, подобна substr
16. compact($1,$2) - Упаковывает в массив переменные и их значения
17. extract($1) - Функция обратная функции compact
18. range($i, $d) - Создает список от $i до $d
19. array_interaect($array1, $array2) - (Пересечение) возвращает значения которые присутствуют в $array1 и $array2
20. array_diff($array1, $array2) - (Разность) Возвращает массив составленый из значений $array1,  не входящих ни в один из массивов 
21. array_unique($array) - Возвращает массив из уникальных значений из массива $array
// JSON
22. json_encode - Преобразует переменную в JSON-последовательность 
23. jsone_decode - Декодирует JSON-последовательность в ассоциотивный массив
// Файлы //
// CSV файлы
1. fgetcsv($fopen,100,';') - Чтение CSV файла, 
$fopen - Идентификатор доступа, 100 - длина чтения, ; - разделитель(смотрится при открытии csv блокнотом)
2. fputcsv($file, $line) - Записывает данные ($line) в CSV файл ($file)
3. fread($file, $numbytes) - Читает из файла блок из символов
4. fwrite($file, $string) или fputs() - Записывает содержимое $string в файл $file
5. fgets($file) или file() - Читает из файла одну строку до /natcasesort
6. basename($path, $suffix) - Выделяет имя файла из пути, если $suffix совпадает
с разрешением файлом, то оно будет отброшенно
7. dirname($path, $level) - Возвращает полное имя каталога, $level - уровень каталога(НЕобязательный)
8. tempnam($dir, $pref) - Генерирует имя каталога с префиком
9. realpath('путь') - Преобразует относительный путь в абсолютный
10. file_get_contents($filename, константы) - Полностью считывает файл в одну строку
11. file_put_contents($filename, $data) - Запись в файл в одно действие
12. parse_ini_file($filename, true) - Читает INI файл
// Каталоги
13. mkdir($name, $root) - Создает каталог с правами доступа
14. rmdir($name) - Удаляет каталог с именем $name
15. chdir($dir) - Сменяет текущий каталог на указанный
16. opendir($path) - Открывает каталог для дальнейшего его считывания
17. readdir($opendir) - считывает информацию по файлам
18. closedir($dir) - Закрывает открытый каталог
19. rewindir($dir) - Перематывает внутренний указатель на начало
20. glob($pattern) - Возвращает список всех путей, подходящую под маску $pattern ('*.txt')
